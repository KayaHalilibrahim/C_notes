#include <stdlib.h>
#include<string.h>

union test  // struct ile aynı  sadece daha az alan kaplıyor
            // buyuk yer kaplayan tür küçük olan ile paylasıyor.....
{
    int x,y;

};


int main()
{

union test t;

t.x=35;
t.y=25;

printf("value of the x:%d value of the y:%d",t.x,t.y);

printf("\n%d",sizeof(t)); // ne kadar yer tuutğunu hesaplamak için..

 return 0;
