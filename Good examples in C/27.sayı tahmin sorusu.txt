#include<stdio.h>
int main()
{
	int secretNumber = 6;  // gizli numara
	int guess;             // tahmin etmek
	int guessCount = 0;    // tahmin sayısı
	int guessLimit = 5;    // tahmin limiti
	int outOfGuesses = 0;  // tahminlerin dışında
	
	while(guess != secretNumber && outOfGuesses == 0) 
	{
		if(guessCount < guessLimit)
		{
			printf("Enter a number :");
			scanf("%d",&guess);
			guessCount++;
		}
		else
		{
			outOfGuesses = 1;
		}	
	}
	if(outOfGuesses == 1)
	{
		printf("\n Out Of Guesses ");
	}
	else
	{
		printf("\n You won!");
	}
	return 0;
	
}
