#include <stdio.h>
#include <stdlib.h>
#define size 5
int main()
{

   int i,max,min;
   int dizi[size];
   int *kaya;

   printf("dizinin elamnalarini giriniz:\n");


   for(kaya=dizi; kaya<dizi+size; kaya++)
   {
       scanf("%d",kaya);
   }



max=min=dizi[0];

for(kaya=dizi+1; kaya<dizi+size; kaya++)
{
    if(*kaya>max)
    {
        max=*kaya;
    }
}

for(kaya=dizi+1; kaya<dizi+size; kaya++)
{
    if(*kaya<min)
    {
        min=*kaya;

    }
}

printf("dizinin en buyuk degeri:%d\n",max);

printf("dizinin en kucuk degeri:%d",min);

return 0;
}
