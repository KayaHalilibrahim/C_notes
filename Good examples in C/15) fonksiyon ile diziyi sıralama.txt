
#include<stdio.h>
#include<stdlib.h>

void siralama(int dizi[],int x)
{

int i,j,gecici;

for ( i=0; i<x-1; i++)
{
  for (j=0; j<x-1-i; j++)
  {
    if (dizi[j]>dizi[j+1])
    {
      gecici=dizi[j];
      dizi[j]=dizi[j+1];
      dizi[j+1]=gecici;
    
  }
  
}

}
}

int main()
{

int x,i;


printf("dizi kac boyutlu olsun:\n");
scanf("%d",&x);

int dizi[x];

srand(time(0));

for (i=0; i<x; i++)
{
  dizi[i]=rand()%70;
  printf("%d\n",dizi[i]);
}

printf("dizinin sirali hali:\n");
siralama(dizi,x);

for (i=0; i<x; i++)
{
  printf("%4d",dizi[i]);
}

  return 0;
}
