#include <stdio.h>
#include <stdlib.h>
#include<math.h>
#define vergi_kitap 4.0
#define vergi_temelgida 5.6
#define vergi_luks 19.6
#define kitap 0
#define temel_gida 1
#define luks 2

int main()
{
int kod;
float fiyat;

printf("lutfen kod ve fiyati giriniz:\n");
scanf("%d  %f",&kod,&fiyat);


if (kod==kitap)
{
    printf("%.1f tl",fiyat+fiyat*(vergi_kitap/100));
}

else if (kod==luks)
{
    printf("%.1f tl",fiyat+fiyat*(vergi_luks/100));
}    

else if (kod==temel_gida)
{
    printf("%.1f tl",fiyat+fiyat*(vergi_temelgida/100));
}  
else
{
    printf("wrongg");
}


    return 0;
}
