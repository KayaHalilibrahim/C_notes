#include <stdio.h>
#include <stdlib.h>

int main()
{
   int arr[]={10,20,30,40,50};

   int *ptr;           //int=>4 bit 

   ptr=arr;

   printf("1.eleman:%d\n",*ptr);
   printf("2.eleman:%d\n",*(ptr+1));
   printf("5.eleman:%d\n",*(ptr+4));
    return 0;
}


output:

1.eleman:10
2.eleman:20
5.eleman:50




/*  örneğin 10=1000 bit ise 
20=1004 ,30=1008 bit yer tutar
burda +1 değeri 4 bit yer tutar!!! */
