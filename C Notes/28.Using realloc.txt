// daha önce malloc calloc ile ayırdığımız alanda güncelleme yapmamızı sağlar.
// örneğin depolama alanı yetersiz gelirse genişletilmesini sağlar..(güncelleme)..

// dosyadan satır satır bilgi okuyup konsola yazdıma


#include <stdio.h>
#include <stdlib.h>

int main()
{

int* ptr;
int n=5;
int i;

ptr=(int*)calloc(n,sizeof(int));

if(ptr==NULL)
{
    printf("didn't become");
    exit(0);
}
else
    {
        for(i=0; i<n; i++)
        {
            ptr[i]=i+1;
        }
        printf("my elements: \n ");

        for(i=0; i<n; i++)
        {
            printf("%d ,",ptr[i]);
        }
        n=10;
        ptr=(int*)realloc (ptr,n*sizeof(int));
        printf("\nsuccesfull\n");

        for(i=5; i<10; i++)
        {
            ptr[i]=i+1;
        }
        for(i=0; i<10; i++)
        {
            printf("%d ,",ptr[i]);
        }
    }
    return 0;
}
